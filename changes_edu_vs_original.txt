+/* base64url encoding (RFC 4648 §5, no padding) */
+static const char b64url_table[64] =
+    "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-_";
+
+static int
+b64url_encode(const unsigned char *in, int inlen, char *out, int outmax)
+{
+    int i, j = 0;
+    for (i = 0; i < inlen; i += 3) {
+        unsigned int b0 = in[i];
+        unsigned int b1 = (i + 1 < inlen) ? (unsigned int)in[i + 1] : 0;
+        unsigned int b2 = (i + 2 < inlen) ? (unsigned int)in[i + 2] : 0;
+        if (j + 4 >= outmax)
+            return -1;
+        out[j++] = b64url_table[b0 >> 2];
+        out[j++] = b64url_table[((b0 & 3) << 4) | (b1 >> 4)];
+        if (i + 1 < inlen)
+            out[j++] = b64url_table[((b1 & 15) << 2) | (b2 >> 6)];
+        if (i + 2 < inlen)
+            out[j++] = b64url_table[b2 & 63];
+    }
+    out[j] = '\0';
+    return j;
+}

+/* Decode application/x-www-form-urlencoded value in-place */
+static void
+url_decode(const char *src, char *dst, int dstmax)
+{
+    int i = 0, j = 0;
+    while (src[i] && j < dstmax - 1) {
+        unsigned char c = (unsigned char)src[i];
+        if (c == '+') {
+            dst[j++] = ' '; i++;
+        } else if (c == '%'
+                && isxdigit((unsigned char)src[i+1])
+                && isxdigit((unsigned char)src[i+2])) {
+            int hi = isdigit((unsigned char)src[i+1]) ? src[i+1]-'0'
+                     : tolower((unsigned char)src[i+1])-'a'+10;
+            int lo = isdigit((unsigned char)src[i+2]) ? src[i+2]-'0'
+                     : tolower((unsigned char)src[i+2])-'a'+10;
+            dst[j++] = (char)((hi << 4) | lo);
+            i += 3;
+        } else {
+            dst[j++] = src[i++];
+        }
+    }
+    dst[j] = '\0';
+}

+/* Write a JSON-escaped string into dst, return chars written */
+static int
+json_esc(const char *src, char *dst, int dstmax)
+{
+    int i = 0, j = 0;
+    while (src[i] && j < dstmax - 2) {
+        unsigned char c = (unsigned char)src[i];
+        if (c == '"' || c == '\\') {
+            if (j + 2 >= dstmax) break;
+            dst[j++] = '\\'; dst[j++] = c;
+        } else if (c < 0x20) {
+            if (j + 7 >= dstmax) break;
+            j += snprintf(dst + j, dstmax - j, "\\u%04x", c);
+        } else {
+            dst[j++] = c; /* pass-through UTF-8 bytes unchanged */
+        }
+        i++;
+    }
+    dst[j] = '\0';
+    return j;
+}

+/* Load dictionary from tab-separated file: en\tja\tother\tother_lang */
+/* Read a JSON string (p points just after opening '"'), write to dst.
+ * Returns pointer after closing '"'. Handles \\ \" \/ \n \t \uXXXX. */
+static const char *
+json_read_str(const char *p, char *dst, int dstmax)
+{
+    int j = 0;
+    while (*p && *p != '"' && j < dstmax - 1) {
+        if (*p == '\\') {
+            p++;
+            switch (*p) {
+            case '"': case '\\': case '/':
+                dst[j++] = *p; break;
+            case 'n': dst[j++] = '\n'; break;
+            case 't': dst[j++] = '\t'; break;
+            case 'r': dst[j++] = '\r'; break;
+            case 'u':
+                if (isxdigit((unsigned char)p[1]) && isxdigit((unsigned char)p[2])
+                 && isxdigit((unsigned char)p[3]) && isxdigit((unsigned char)p[4])) {
+                    unsigned int cp;
+                    char hex[5] = { p[1], p[2], p[3], p[4], '\0' };
+                    cp = (unsigned int)strtoul(hex, NULL, 16);
+                    p += 4;
+                    if (cp < 0x80 && j < dstmax - 1) {
+                        dst[j++] = (char)cp;
+                    } else if (cp < 0x800 && j < dstmax - 2) {
+                        dst[j++] = (char)(0xC0 | (cp >> 6));
+                        dst[j++] = (char)(0x80 | (cp & 0x3F));
+                    } else if (j < dstmax - 3) {
+                        dst[j++] = (char)(0xE0 | (cp >> 12));
+                        dst[j++] = (char)(0x80 | ((cp >> 6) & 0x3F));
+                        dst[j++] = (char)(0x80 | (cp & 0x3F));
+                    }
+                }
+                break;
+            default: dst[j++] = *p; break;
+            }
+        } else {
+            dst[j++] = *p;
+        }
+        p++;
+    }
+    dst[j] = '\0';
+    if (*p == '"') p++;
+    return p;
+}

+            while (*p == ' ' || *p == ':') p++;

+/* Serialize dictionary to JSON object string (caller must free) */
+static char *

+{

+        buf[pos++] = '"';

+static void
+http_handle(SOCKET c)
+{
+    char req[8192], method[8], path_buf[512];
+    const char *body;
+    DWORD timeout = 2000;
+    int n;
+
+    setsockopt(c, SOL_SOCKET, SO_RCVTIMEO, (char *)&timeout, sizeof(timeout));
+    n = recv(c, req, sizeof(req) - 1, 0);
+    if (n <= 0) return;
+    req[n] = '\0';
+    if (sscanf(req, "%7s %511s", method, path_buf) != 2) return;
+    { char *q = strchr(path_buf, '?'); if (q) *q = '\0'; }
+
+    body = strstr(req, "\r\n\r\n");
+    body = body ? body + 4 : "";
+
+    /* GET /data */
+    if (strcmp(method, "GET") == 0
+            && strncmp(path_buf, "/data", 5) == 0
+            && (path_buf[5] == '\0' || path_buf[5] == '?')) {
+        char *buf = (char *)malloc(HTTP_DMAX);
+        int   len = 0;
+        InterlockedExchange(&s_viewer_connected, 1); /* viewer is open */
+        if (buf) {
+            EnterCriticalSection(&s_screen_cs);
+            len = (int)strlen(s_screen_buf);
+            memcpy(buf, s_screen_buf, len);
+            LeaveCriticalSection(&s_screen_cs);
+            http_respond(c, 200, "text/plain; charset=US-ASCII", buf, len);
+            free(buf);
+        }
+        return;
+    }
+
+    /* GET /dict.json  – serve dictionary as JSON */
+    if (strcmp(method, "GET") == 0
+            && strncmp(path_buf, "/dict.json", 10) == 0) {
+        int jlen;
+        char *jbuf = dict_to_json(&jlen);
+        if (jbuf) {
+            http_respond(c, 200, "application/json; charset=UTF-8", jbuf, jlen);
+            free(jbuf);
+        } else {
+            http_respond(c, 200, "application/json", "{}", 2);
+        }
+        return;
+    }
+
+    /* POST /dict.json  – add/update dictionary entry */
+    if (strcmp(method, "POST") == 0
+            && strncmp(path_buf, "/dict.json", 10) == 0) {
+        char en[DICT_EN], ja[DICT_VAL], other[DICT_VAL], lang[DICT_LANG];
+        char resp[64];
+        int  rlen;
+        form_field(body, "en",         en,    sizeof(en));
+        form_field(body, "ja",         ja,    sizeof(ja));
+        form_field(body, "other",      other, sizeof(other));
+        form_field(body, "other_lang", lang,  sizeof(lang));
+        if (en[0] && (ja[0] || other[0])) {
+            dict_set(en, ja, other, lang);
+            dict_save_file();
+            rlen = snprintf(resp, sizeof(resp),
+                            "{\"ok\":true,\"count\":%d}", s_dict_cnt);
+        } else {
+            rlen = snprintf(resp, sizeof(resp),
+                            "{\"ok\":false,\"error\":\"missing fields\"}");
+        }
+        http_respond(c, 200, "application/json", resp, rlen);
+        return;
+    }
+
+    /* GET /  or  /nethack_viewer.html  – serve the HTML file */
+    if (strcmp(method, "GET") == 0
+            && (strcmp(path_buf, "/") == 0
+                || strncmp(path_buf, "/nethack_viewer.html", 20) == 0)) {
+        FILE *fp = fopen(s_html_path, "rb");
+        if (!fp) {
+            const char *msg = "nethack_viewer.html not found";
+            http_respond(c, 404, "text/plain", msg, (int)strlen(msg));
+            return;
+        }
+        fseek(fp, 0, SEEK_END);
+        long fsz = ftell(fp);
+        rewind(fp);
+        char *fbuf = (char *)malloc(fsz);
+        if (fbuf) {
+            fread(fbuf, 1, fsz, fp);
+            http_respond(c, 200, "text/html; charset=UTF-8", fbuf, (int)fsz);
+            free(fbuf);
+        }
+        fclose(fp);
+        return;
+    }
+
+    http_respond(c, 404, "text/plain", "Not Found", 9);
+}

+static DWORD WINAPI
+http_server_thread(LPVOID unused)
+{
+    WSADATA            wsa;
+    SOCKET             srv, cli;
+    struct sockaddr_in addr;
+    int                yes = 1;
+
+    (void)unused;
+    if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0) return 1;
+    srv = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
+    if (srv == INVALID_SOCKET) { WSACleanup(); return 1; }
+    setsockopt(srv, SOL_SOCKET, SO_REUSEADDR, (char *)&yes, sizeof(yes));
+    memset(&addr, 0, sizeof(addr));
+    addr.sin_family      = AF_INET;
+    addr.sin_addr.s_addr = htonl(INADDR_LOOPBACK);
+    addr.sin_port        = htons((u_short)s_http_port);

+    if (!newbuf) { free(buf); return; }

+    memcpy(newbuf + prefix_len,           newurl, new_len);

+    free(buf);

+/* -------------------------------------------------------------------------
+ * Binary format sent to /data (base64url encoded):
+ *   Byte 0   : screen width  (uint8)
+ *   Byte 1   : screen height (uint8)
+ *   Byte 2   : cursor X      (uint8)
+ *   Byte 3   : cursor Y      (uint8)
+ *   Byte 4.. : RLE cell stream
+ *     Each run: [count-1 (uint8)] [cp_lo (uint8)] [cp_hi (uint8)] [attr (uint8)]
+ * ------------------------------------------------------------------------- */
+static void
+dump_screen_url(void)
+{
+    int total, i, outpos, b64len;
+    unsigned char *binbuf;
+    char *b64buf;
+    size_t binmax, b64max;
+
+    if (!console.back_buffer || console.width <= 0 || console.height <= 0)
+        return;
+
+    total  = console.width * console.height;
+    binmax = 5 + (size_t)total * 4;   /* +1 for flags byte */
+    b64max = (binmax * 4 / 3) + 8;
+
+    binbuf = (unsigned char *)malloc(binmax);
+    b64buf = (char *)malloc(b64max);
+    if (!binbuf || !b64buf) {
+        free(binbuf); free(b64buf); return;
+    }
+
+    binbuf[0] = (unsigned char)console.width;
+    binbuf[1] = (unsigned char)console.height;
+    binbuf[2] = (unsigned char)console.cursor.X;
+    binbuf[3] = (unsigned char)console.cursor.Y;
+    /* byte 4: flags  bit0=getlin mode (suppress per-char speech during line input) */
+    binbuf[4] = s_getlin_active ? 1 : 0;
+    outpos = 5;
+
+    i = 0;
+    while (i < total) {
+        cell_t *c = &console.back_buffer[i];
+        unsigned short cp =
+            (c->character == 0xFFFF || c->character == 0)
+                ? 32 : (unsigned short)c->character;
+        unsigned char attr = (unsigned char)(c->attribute & 0xFF);
+        int run = 1;
+        while (run < 256 && i + run < total) {
+            cell_t *nc = &console.back_buffer[i + run];
+            unsigned short ncp =
+                (nc->character == 0xFFFF || nc->character == 0)
+                    ? 32 : (unsigned short)nc->character;
+            unsigned char nattr = (unsigned char)(nc->attribute & 0xFF);
+            if (ncp != cp || nattr != attr) break;
+            run++;
+        }
+        binbuf[outpos++] = (unsigned char)(run - 1);
+        binbuf[outpos++] = (unsigned char)(cp & 0xFF);
+        binbuf[outpos++] = (unsigned char)((cp >> 8) & 0xFF);
+        binbuf[outpos++] = attr;
+        i += run;
+    }
+
+    b64len = b64url_encode(binbuf, outpos, b64buf, (int)b64max);
+    free(binbuf);
+    if (b64len >= 0)
+        http_data_set(b64buf, b64len);
+    free(b64buf);
+}
